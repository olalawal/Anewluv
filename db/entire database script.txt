CREATE PROCEDURE [dbo].[sp_MemberPermanently]
	-- Add the parameters for the stored procedure here
@S_No
int 
	
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;


   --delete mail stuff
   delete from mailboxmessages where recipient_id = @S_No 
      delete from mailboxmessages where sender_id = @S_No 
    
    --delete activities
    delete from profileactivities where profile_id  =@S_No
    
    --delete actions to and from
   	 DELETE FROM actions
WHERE creator_profile_id =@S_No

   	 DELETE FROM actions
WHERE target_profile_id =@S_No
   
   --now delete meta data 
	 DELETE FROM profilemetadatas
WHERE profile_id =@S_No

DELETE FROM profiledatas


WHERE profile_id = @S_No

--finally delete the profile now
delete from profiles where id =@S_No

END
