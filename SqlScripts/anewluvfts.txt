CREATE FUNCTION [dbo].[fnTruncateString]
	(
	
	@StringToTruncate  Varchar(MAX),
	@LengthToTrunCateBy int
	
	)
RETURNS VARCHAR(MAX)

AS
	BEGIN
	declare @length int;

	set @length = LEN(@StringToTruncate) 
	
	IF @length > @LengthToTrunCateBy
	BEGIN
		RETURN SUBSTRING(@StringToTruncate, 0, @LengthToTrunCateBy)
	END
	    RETURN @StringToTruncate
	END
GO

    Declare @DateThreeMonthsAgo DateTime; 
    Declare @DateSixMonthsAgo DateTime;
    Declare @DateOneMonthAgo DateTime; 
    Declare @ReturnText Varchar(25);
    
   Set @Today = GetDate();    
      --Subtract 1 days
   Set @DateToday = DATEADD(HOUR, -23, @Today);
      --Subtract 2 days
   Set @DateYesterday = DATEADD(HOUR, -36, @Today);   
    --Subtract 3 days
   Set @DateThreeDaysAgo = DATEADD(day, -3, @Today);
   --Subtract 1 weeks               
   Set @DateOneWeekAgo = DATEADD(day, -7, @Today);
    --Subtract 3 weeks   
   Set @DateThreeWeeksago = DATEADD(day, -21, @Today);   
     --Subtract 4 weeks =1 months      
   Set @DateOneMonthAgo = DATEADD(month, -1, @Today);
   --Subtract 12 weeks =3 months        
   Set @DateThreeMonthsAgo = DATEADD(month, -3, @Today);
   --Subtract 24 weeks =6 months   
   Set @DateSixMonthsAgo = DATEADD(month, -6, @Today);
   
   SET @ReturnText = 
     CASE 
		 WHEN (@lastloggedintime > @DateToday) THEN  'Today'
		 WHEN (@lastloggedintime > @DateYesterday) THEN  'Two Days Ago'
         WHEN (@lastloggedintime > @DateThreeDaysAgo) THEN  'Last Three Days'
         WHEN (@lastloggedintime > @DateOneWeekAgo) THEN   'Last Week '
         WHEN (@lastloggedintime > @DateThreeWeeksago) THEN  'Three Weeks Ago'
         WHEN (@lastloggedintime > @DateOneMonthAgo) THEN  'One Month Ago'
       --  WHEN (@lastloggedintime > @DateThreeMonthsAgo) THEN  'O Months Ago'
         WHEN (@lastloggedintime > @DateSixMonthsAgo) THEN  'Six Months Ago'        
         ELSE 'Over one Month'
      END

   
   
  Return @ReturnText;




END

create Function [dbo].[fnGetDistance] 
( 
      @Lat1 Float(18),  
      @Long1 Float(18), 
      @Lat2 Float(18), 
      @Long2 Float(18), 
      @ReturnType VarChar(10) 
) 




Returns Float(18) 




AS 




Begin 




      Declare @R Float(8); 
      Declare @dLat Float(18); 
      Declare @dLon Float(18); 
      Declare @a Float(18); 
      Declare @c Float(18); 
      Declare @d Float(18); 


      Set @R =  
            Case @ReturnType  
            When 'Miles' Then 3956.55  
            When 'Kilometers' Then 6367.45 
            When 'Feet' Then 20890584 
            When 'Meters' Then 6367450 
            Else 20890584 -- Default feet (Garmin rel elev) 
            End 




      Set @dLat = Radians(@lat2 - @lat1); 




      Set @dLon = Radians(@long2 - @long1); 




      Set @a = Sin(@dLat / 2)  
                 * Sin(@dLat / 2)  
                 + Cos(Radians(@lat1)) 
                 * Cos(Radians(@lat2))  
                 * Sin(@dLon / 2)  
                 * Sin(@dLon / 2); 
      Set @c = 2 * Asin(Min(Sqrt(@a))); 

      Set @d = @R * @c; 
      Return @d; 

End
